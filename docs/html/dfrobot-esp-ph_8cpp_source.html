80float DFRobotESPpH::readPH(float voltage, float temperature) {
81
82 float slope = (7.0 - 4.0) / ((this->_neutralVoltage - 1500.0) / 3.0 - (this->_acidVoltage - 1500.0) / 3.0);
83 float intercept = 7.0 - slope * (this->_neutralVoltage - 1500.0) / 3.0;
84 this->_phValue = slope * (voltage - 1500.0) / 3.0 + intercept;
85 return _phValue;
86}

132boolean DFRobotESPpH::cmdSerialDataAvailable()
133{
134 char cmdReceivedChar;
135 static unsigned long cmdReceivedTimeOut = millis();
136 while (Serial.available() > 0)
137 {
138 if (millis() - cmdReceivedTimeOut > 500U)
139 {
140 this->_cmdReceivedBufferIndex = 0;
141 memset(this->_cmdReceivedBuffer, 0, (ReceivedBufferLength));
142 }
143 cmdReceivedTimeOut = millis();
144 cmdReceivedChar = Serial.read();
145 if (cmdReceivedChar == '\n' || this->_cmdReceivedBufferIndex == ReceivedBufferLength - 1)
146 {
147 this->_cmdReceivedBufferIndex = 0;
148 strupr(this->_cmdReceivedBuffer);
149 return true;
150 }
151 else
152 {
153 this->_cmdReceivedBuffer[this->_cmdReceivedBufferIndex] = cmdReceivedChar;
154 this->_cmdReceivedBufferIndex++;
155 }
156 }
157 return false;
158}

194void DFRobotESPpH::phCalibration(byte mode) {
195 char *receivedBufferPtr;
196 static boolean phCalibrationFinish = 0;
197 static boolean enterCalibrationFlag = 0;
198 switch (mode)
199 {
200 case 0:
201 if (enterCalibrationFlag)
202 {
203 Serial.println(F(">>>Command Error<<<"));
204 }
205 break;
206
207 case 1:
208 enterCalibrationFlag = 1;
209 phCalibrationFinish = 0;
210 Serial.println();
211 Serial.println(F(">>>Enter PH Calibration Mode<<<"));
212 Serial.println(F(">>>Please put the probe into the 4.0 or 7.0 standard buffer solution<<<"));
213 Serial.println();
214 break;
215
216 case 2:
217 if (enterCalibrationFlag)
218 {
219 if ((this->_voltage > PH_8_VOLTAGE) && (this->_voltage < PH_6_VOLTAGE))
220 { // buffer solution:7.0
221 Serial.println();
222 Serial.print(F(">>>Buffer Solution:7.0"));
223 this->_neutralVoltage = this->_voltage;
224 Serial.println(F(",Send EXITPH to Save and Exit<<<"));
225 Serial.println();
226 phCalibrationFinish = 1;
227 }
228 else if ((this->_voltage > PH_5_VOLTAGE) && (this->_voltage < PH_3_VOLTAGE))
229 { //buffer solution:4.0
230 Serial.println();
231 Serial.print(F(">>>Buffer Solution:4.0"));
232 this->_acidVoltage = this->_voltage;
233 Serial.println(F(",Send EXITPH to Save and Exit<<<"));
234 Serial.println();
235 phCalibrationFinish = 1;
236 }
237 else
238 {
239 Serial.println();
240 Serial.print(F(">>>Buffer Solution Error Try Again<<<"));
241 Serial.println(); // not buffer solution or faulty operation
242 phCalibrationFinish = 0;
243 }
244 }
245 break;
246
247 case 3://store calibration value in eeprom
248 if (enterCalibrationFlag)
249 {
250 Serial.println();
251 preferences.begin("pHVals", false);
252 if (phCalibrationFinish)
253 {
254 if ((this->_voltage > PH_8_VOLTAGE) && (this->_voltage < PH_5_VOLTAGE))
255 {
256 preferences.putFloat("voltage7", this->_neutralVoltage);
257 Serial.print(F("PH 7 Calibration value SAVE THIS FOR LATER: "));
258 Serial.print(this->_neutralVoltage);
259 }
260 else if ((this->_voltage > PH_5_VOLTAGE) && (this->_voltage < PH_3_VOLTAGE))
261 {
262 preferences.putFloat("voltage4", this->_acidVoltage);
263 Serial.print(F("PH 4 Calibration value SAVE THIS FOR LATER: "));
264 Serial.print(this->_acidVoltage);
265 }
266 Serial.print(F(">>>Calibration Successful"));
267 }
268 else
269 {
270 Serial.print(F(">>>Calibration Failed"));
271 }
272 preferences.end();
273 Serial.println(F(",Exit PH Calibration Mode<<<"));
274 Serial.println();
275 phCalibrationFinish = 0;
276 enterCalibrationFlag = 0;
277 }
278 break;
279 }
280}
