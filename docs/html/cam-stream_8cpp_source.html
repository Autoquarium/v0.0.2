84 live_client.flush();
85 live_client.print("\n");

89void CamStream::http_resp() {
90 WiFiClient client = server.available();
91 if (client.connected()) {
92 String req = "";
93 while(client.available()) {
94 req += (char)client.read();
95 }
96 Serial.println("request " + req);
97 int addr_start = req.indexOf("GET") + strlen("GET");
98 int addr_end = req.indexOf("HTTP", addr_start);
99 if (addr_start == -1 || addr_end == -1) {
100 Serial.println("Invalid request " + req);
101 return;
102 }
103 req = req.substring(addr_start, addr_end);
104 req.trim();
105 Serial.println("Request: " + req);
106 client.flush();
107
108 String s;
109 if (req == "/") {
110 s = "HTTP/1.1 200 OK\n";
111 s += "Content-Type: text/html\n\n";
112 s += index_html;
113 s += "\n";
114 client.print(s);
115 client.stop();
116 }
117 else if (req == "/video") {
118 live_client = client;
119 live_client.print("HTTP/1.1 200 OK\n");
120 live_client.print("Content-Type: multipart/x-mixed-replace; boundary=frame\n\n");
121 live_client.flush();
122 connected = true;
123 }
124 else {
125 s = "HTTP/1.1 404 Not Found\n\n";
126 client.print(s);
127 client.stop();
128 }
129 }
130}
