60void FishMqtt::MQTTreconnect() {
61 Serial.println("Connecting to MQTT Broker...");
62 while (!connected()) {
63 Serial.println("Reconnecting to MQTT Broker..");
64 if (connect(clientName, usrname, password)) {
65 //Serial.println("Connected to broker.");
66 //subscribe to topic

68 char topic_c[40];

70 subscribe(topic_c); //subscribes to all the commands messages triggered by the user
71 Serial.print("Subscribed to topic: ");

73 return;
74 }
75 if(WiFi.status() != WL_CONNECTED) {
